91           [HarmonyPatch]
92           class LocalizationManagerPatch
93           {
94               static MethodBase TargetMethod()
95               {
96                   return typeof(LocalizationManager).GetMethod("Load");
97               }
98   
99               static IEnumerator ProcessLocalizationManager()
100              {
101                  while (
102                      ReferenceEquals(LocalizationManager.instance, null) || 
103                      ReferenceEquals(LocalizationManager.instance.textTable, null) ||
104                      ReferenceEquals(LocalizationManager.instance.textTable.entries, null)
105                  )
106                  {
107                      yield return new WaitForEndOfFrame();
108                  }
109  
110                  var clone = new Dictionary<string, string>(LocalizationManager.instance.textTable.count);
111                  
112                  foreach (var entry in LocalizationManager.instance.textTable)
113                  {
114                      //Log(ConCol.Debug, "L:" + LocalizationManager.instance.activeLanguage + " || K:" + entry.key + " || V:" + entry.value);
115                      try
116                      {
117                          if (!ReferenceEquals(LocaleOverrides[entry.Key], null))
118                          {
119                              clone[entry.Key] = LocaleOverrides[entry.Key];
120                          }
121                          else
122                          {
123                              clone[entry.Key] = entry.Value;
124                          }
125                      }catch{}
126                  }
127                  LocalizationManager.instance.textTable = clone;
128              }
129              
130              static void Postfix()
131              {
132                  MelonCoroutines.Start(ProcessLocalizationManager());
133              }
134          }
